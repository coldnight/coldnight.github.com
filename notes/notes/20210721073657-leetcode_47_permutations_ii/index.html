LeetCode: 47. Permutations II

July 21, 2021

在 LeetCode: 46. Permutations 的基础上增加重复的元素。感觉不能依赖于 track + map 的去重逻辑回溯。

class Solution {
public:
    vector<vector<int>> permuteUnique(vector<int>& nums) {
        vector<vector<int>> res;
        backtrack(nums);
        return res;
    }

    void backtrack(vector<int>& nums) {
        int n = 0;
        int sz = nums.size();

        for (int i = 0; i < sz; i++) {
            n = nums.back();
            nums.pop_back();
            cout << n << ", ";
            backtrack(nums);
            nums.push_back(n);
        }
    }
};
